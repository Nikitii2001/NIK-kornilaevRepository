#include <iostream>
#include <stdio.h>
#include <cmath>

using namespace std;
int gcd(int a, int b){
int a1=abs(a);
int b1=abs(b);
while(b1!=0){
a1%=b1;
swap(a1,b1);
}
return a1;
}
int lcm(int a, int b){
return a*b/gcd(a,b);
}
class drob {
private:
int ch, zn;
public:
drob(){
ch=0;
zn=0;
}
drob(int a, int b){
ch=a;
zn=b;
}
drob(int a, int b, int c){
ch=a*c+b;
zn=c;
}
void write(){
int a=gcd(ch,zn);
if(a!=1){
ch/=a;
zn/=a;
}
if(abs(ch)>zn){
int c=ch/zn;
ch=abs(ch)-(zn*abs(c));
cout << "(" << c << ")";
}
cout << ch << "/" << zn << endl;
}
int getch(){
return ch;
}
int getzn(){
return zn;
}
void setch(int a){
this->ch=a;
}
void setzn(int a){
this->zn=a;
}
};
int main(){
    drob a(2, 6), b(11, 12);
    a.write();
    cout << endl;
    b.write();

}
